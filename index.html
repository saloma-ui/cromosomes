    const a = arr.slice();

    for (let i = a.length - 1; i > 0; i--) {

      const j = Math.floor(Math.random() * (i + 1));

      [a[i], a[j]] = [a[j], a[i]];

    }

    return a;

  function renderQuestion(){

      btn.appendChild(span);

 

      btn.addEventListener("click", () => choose(opt.key, opt.text, shuffledOptions));

 

      elOptions.appendChild(btn);

    });

  }

 

  function choose(chosenKey, chosenText, shuffledOptions){

    if (answered) return;

    answered = true;

 

    const q = questions[order[current]];

    const ok = chosenKey === q.answerKey;

 

    // Deshabilita tots els botons i marca correcte/incorrecte

    const buttons = Array.from(elOptions.querySelectorAll("button.option"));

    buttons.forEach((b) => {

      b.disabled = true;

      const k = b.dataset.key;

      if (k === q.answerKey){

        b.style.borderColor = "rgba(54,211,153,0.65)";

        b.style.background = "rgba(54,211,153,0.16)";

      }

      if (k === chosenKey && !ok){

        b.style.borderColor = "rgba(248,114,114,0.70)";

        b.style.background = "rgba(248,114,114,0.14)";

      }

    });

 

    if (ok){

      score += 1;

      correct += 1;

      elFeedback.classList.add("show", "good");

      elFb1.textContent = "Correcte ✅";

      elFb2.textContent = q.explain;

      elStatusBadge.textContent = "Resposta correcta";

      elStatusBadge.style.borderColor = "rgba(54,211,153,0.55)";

    } else {

      elFeedback.classList.add("show", "bad");

      elFb1.textContent = "Incorrecte ❌";

      const correctOpt = q.options.find(o => o.key === q.answerKey);

      elFb2.textContent = `La correcta era: “${correctOpt.text}”. ${q.explain}`;

      elStatusBadge.textContent = "Resposta incorrecta";

      elStatusBadge.style.borderColor = "rgba(248,114,114,0.55)";

    }

 

    // Log per revisió final

    const correctText = q.options.find(o => o.key === q.answerKey).text;

    reviewLog.push({

      qText: q.text,

      chosenText,

      correctText,

      ok,

      explain: q.explain

    });

 

    elScore.textContent = String(score);

    elCorrect.textContent = String(correct);

 

    elNext.disabled = false;

    elNext.focus();

    updateProgressAfterAnswer();

  }

 

  function updateProgressAfterAnswer(){

    const pct = ((current + 1) / questions.length) * 100;

    elProgress.style.width = pct.toFixed(0) + "%";

  }

 

  function next(){

    if (!answered) return;

    if (current < questions.length - 1){

      current += 1;

      renderQuestion();

    } else {

      showResults();

    }

  }

        const item = document.createElement("div");

        item.className = "reviewItem";

        item.innerHTML = `

          <div class="rq">${escapeHtml(r.qText)}</div>

          <div class="ra">

            <div><b>Has triat:</b> ${escapeHtml(r.chosenText)}</div>

            <div><b>Correcta:</b> ${escapeHtml(r.correctText)}</div>

            <div style="margin-top:6px;">${escapeHtml(r.explain)}</div>

          </div>

        `;

        elReview.appendChild(item);

      });

    }

 

    elResults.classList.add("show");

  }

 

  function escapeHtml(str){

    return String(str)

      .replaceAll("&", "&amp;")

      .replaceAll("<", "&lt;")

      .replaceAll(">", "&gt;")

      .replaceAll('"', "&quot;")

      .replaceAll("'", "&#039;");

  }

    const key = e.key.toLowerCase();

    if (key === "n") {

      if (!elNext.disabled) next();

      return;

    }

    if (["1","2","3","4"].includes(e.key)) {

      if (answered) return;

      const idx = Number(e.key) - 1;

      const btns = Array.from(elOptions.querySelectorAll("button.option"));

      if (btns[idx]) btns[idx].click();

    }

  });

 

  // Botons

  elNext.addEventListener("click", next);

  elRestart.addEventListener("click", resetGame);

 

  // Inici

  resetGame();

})();
